```csharp
private static void JumpInput()
{
    // 사용자가 점프 키를 누르고 있는지 확인
    if (Input.GetButton("Jump"))
    {
        // Jump 키를 누르고 있다면, 점프 힘을 누적
        // Debug.Log("JumpHold");
        GameManager.Instance.Player.GetComponent<Player>().ChargeJumpForce();
    }
    // 점프 키를 떼면 누적된 힘 만큼 점프 후 점프 힘을 초기화
    if (Input.GetButtonUp("Jump"))
    {
        // Debug.Log("JumpRelease");
        GameManager.Instance.Player.GetComponent<Player>().Jump();
    }
}
```

1. 점프 + 방향키 조합으로 캐릭터를 조작할 수 있다.
