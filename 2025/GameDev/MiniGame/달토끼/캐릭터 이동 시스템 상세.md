```csharp
private static void JumpInput()
{
    // 사용자가 점프 키를 누르고 있는지 확인
    if (Input.GetButton("Jump"))
    {
        // Jump 키를 누르고 있다면, 점프 힘을 누적
        // Debug.Log("JumpHold");
        GameManager.Instance.Player.GetComponent<Player>().ChargeJumpForce();

    // 점프 키를 떼면 누적된 힘 만큼 점프 후 점프 힘을 초기화
    if (Input.GetButtonUp("Jump"))
    {
        // Debug.Log("JumpRelease");
        GameManager.Instance.Player.GetComponent<Player>().Jump();
    }
}
```

```csharp
private static void DirectionInput()
    {
        // 수평 입력 처리
        var horizontal = Input.GetAxis("Horizontal");

        // 방향 벡터 생성
        var direction = new Vector2(horizontal, 0).normalized;

        // 방향 벡터가 유효한 경우에만 처리
        if (direction.magnitude >= 0.1f)
        {
            // 방향 벡터를 사용하여 플레이어 객체의 방향 전환
            GameManager.Instance.Player.GetComponent<Player>().Move(direction);
        }
    }
```

```csharp
public void Move(Vector2 movement)
{
    // 캐릭터가 바라보는 방향으로 회전
    if (movement.x > 0)
    {
        objBody.transform.localScale = new Vector3(1, 1, 1);
    }
    else if (movement.x < 0)
    {
        objBody.transform.localScale = new Vector3(-1, 1, 1);
    }
    
    // 캐릭터가 점프 상태일 때 이동 처리
    if (!IsGrounded())
    {
        if (_rb is null)
        {
            Debug.LogError("Rigidbody is not initialized.");
            return;
        }
    
        _rb.velocity = new Vector2(movement.x * moveSpeed, _rb.velocity.y);
    }
    
}
```

```csharp
private void DirectionInput()
    {
        // 수평 입력 처리
        var horizontal = Input.GetAxis("Horizontal");

        // 방향 벡터 생성
        var direction = new Vector2(horizontal, 0).normalized;

        // 방향 벡터가 유효한 경우에만 처리
        if (direction.magnitude >= 0.1f)
        {
            // 방향 벡터를 사용하여 플레이어 객체의 방향 전환
            if (direction.x < 0)
            {
                _currentDirection = Direction.Left;
                GameManager.Instance.Player.GetComponent<Player>().SetDirection(Direction.Left);
            }
            else if (direction.x > 0)
            {
                _currentDirection = Direction.Right;
                GameManager.Instance.Player.GetComponent<Player>().SetDirection(Direction.Right);
            }
        }
        else
        {
            _currentDirection = Direction.Center;
        }
    }
```
