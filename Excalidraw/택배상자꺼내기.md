#include <string>
#include <vector>
#include <stack>
#include <cmath>

using namespace std;

int solution(int n, int w, int num) {
    int answer = 0;
    vector<stack<int>> truck;
    int box = 1, index = 0;
    bool pivot = false;

    while (box <= n)
    {
        if (index >= w)
        {
            pivot = true;
            index--;
        }
        else if (index < 0)
        {
            pivot = false;
            index++;
        }
        truck[index].push(box);
        if (pivot)
            index--;
        else
            index++;
        box++;
    }

    index = 0;
    float dump = fmod ( (float) num / (float) w, 2.0f );
    
    if ( 0.0f < dump && dump <= 1.0f )
        index = num - 1 % w;
    else
        index = w - (num % w);

    int boxValue = 0;
    do 
    {
        if (truck[index].empty())
            return 0;
        boxValue = truck[index].top();
        truck[index].pop();
        answer++;
    } while(boxValue != num);
    
    return answer;
}
